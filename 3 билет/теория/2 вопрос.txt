<!-- Это комментарии, они не влияют на работу кода -->
<!DOCTYPE html> 
    <!-- Это объявление типа документа. В данном случае указывает, что это документ HTML5 -->

<html lang="en"> 
    <!-- 
        Это открывающий тег элемента `<html>`, 
        который обозначает начало HTML документа. `lang="en"` 
        указывает на язык документа (в данном случае английский), 
        что помогает поисковым системам и другим инструментам понять язык контента 
    -->

<head> <!-- Тег `<head>` содержит информацию о документе, такую как мета-теги, скрипты, стили и заголовки страницы -->

    <meta charset="UTF-8"> 
    <!-- 
        Мета-тег, который указывает кодировку документа. 
        Здесь устанавливается кодировка UTF-8, 
        которая поддерживает большинство языков и символов Unicode 
    -->

    <meta name="viewport" content="width=device-width, initial-scale=1.0"> 
    <!-- 
        Еще один мета-тег, 
        который определяет поведение просмотра страницы на мобильных устройствах. 
        Здесь устанавливаются ширина вида для устройства и начальный масштаб просмотра 
    -->

    <title>Document</title>
    <!--
        Тег `<title>` используется для задания заголовка страницы, 
        который отображается в заголовке браузера или на закладке
    -->
</head> <!-- Закрывающий тег `</head>`, который обозначает конец блока заголовка веб-страницы -->
<body> 
    <!-- 
        Тег `<body>` обозначает начало тела веб-страницы, 
        где размещается основной контент, такой как текст, изображения, ссылки и другие элементы 
    -->
    
</body> <!-- Закрывающий тег `</body>`, который обозначает конец блока тела веб-страницы -->
</html> <!-- Закрывающий тег `</html>`, который обозначает конец HTML документа -->
